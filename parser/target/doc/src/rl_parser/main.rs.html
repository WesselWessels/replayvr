1use boxcars::{Attribute, ParserBuilder, Replay};
2use serde::Serialize;
3use std::collections::HashMap;
4use std::env;
5use std::fs;

7#[derive(Serialize, Clone)]
8struct Vec3 {
9    x: f32,
10    y: f32,
11    z: f32,
12}

14#[derive(Serialize, Clone)]
15struct CarState {
16    id: i32,
17    name: String,
18    team: u8,
19    x: f32,
20    y: f32,
21    z: f32,
22    yaw: f32,
23    boost: u8,
24}
25
26#[derive(Serialize)]
27struct Frame {
28    time: f32,
29    ball: Option<Vec3>,
30    cars: Vec<CarState>,
31}
32
33#[derive(Serialize)]
34struct PlayerMeta {
35    name: String,
36    team: u8,
37}
38
39#[derive(Serialize)]
40struct ReplayMeta {
41    team0_score: i32,
42    team1_score: i32,
43    duration: f32,
44    players: Vec<PlayerMeta>,
45}
46
47#[derive(Serialize)]
48struct Output {
49    meta: ReplayMeta,

52
53fn quat_to_yaw(x: f32, y: f32, z: f32, w: f32) -> f32 {
54    f32::atan2(2.0 * (w * z + x * y), 1.0 - 2.0 * (y * y + z * z))
55}
56
57fn main() {
58    let args: Vec<String> = env::args().collect();
59    let path = args.get(1).expect("Usage: rl-parser <replay_file>");
60
61    let data = fs::read(path).expect("Could not read replay file");
62
63    let replay: Replay = ParserBuilder::new(&data)
64        .on_error_check_crc()
65        .parse()
66        .expect("Could not parse replay");
67
68    let props = &replay.properties;
69
70    let team0_score = props
71        .iter()
72        .find(|(k, _)| k == "Team0Score")
73        .and_then(|(_, v)| if let boxcars::HeaderProp::Int(n) = v { Some(*n) } else { None })
74        .unwrap_or(0);
75
76    let team1_score = props
77        .iter()
78        .find(|(k, _)| k == "Team1Score")
79        .and_then(|(_, v)| if let boxcars::HeaderProp::Int(n) = v { Some(*n) } else { None })
80        .unwrap_or(0);
81
82    let duration = props
83        .iter()
84        .find(|(k, _)| k == "TotalSecondsPlayed")
85        .and_then(|(_, v)| if let boxcars::HeaderProp::Float(f) = v { Some(*f) } else { None })
86        .unwrap_or(0.0);
87
88    let players: Vec<PlayerMeta> = props
89        .iter()
90        .find(|(k, _)| k == "PlayerStats")
91        .and_then(|(_, v)| {
92            if let boxcars::HeaderProp::Array(rows) = v {
93                Some(rows.iter().filter_map(|row| {
94                    let name = row.iter().find(|(k, _)| k == "Name")
95                        .and_then(|(_, v)| if let boxcars::HeaderProp::Str(s) = v { Some(s.clone()) } else { None })?;
96                    let team = row.iter().find(|(k, _)| k == "Team")
97                        .and_then(|(_, v)| if let boxcars::HeaderProp::Int(n) = v { Some(*n as u8) } else { None })
98                        .unwrap_or(0);
99                    Some(PlayerMeta { name, team })
100                }).collect())
101            } else { None }
102        })
103        .unwrap_or_default();
104
105    let network = match replay.network_frames {
106        Some(ref nf) => nf,
107        None => { eprintln!("No network frames"); return; }
108    };
109
110    // actor_id → object name
111    let mut actor_class: HashMap<i32, String> = HashMap::new();
112    let mut ball_pos: Option<Vec3> = None;
113    let mut cars: HashMap<i32, CarState> = HashMap::new();
114    // PRI actor_id → player name
115    let mut pri_name: HashMap<i32, String> = HashMap::new();
116    // PRI actor_id → team (0 or 1)
117    let mut pri_team: HashMap<i32, u8> = HashMap::new();
118    // car actor_id → PRI actor_id
119    let mut car_pri: HashMap<i32, i32> = HashMap::new();
120
121    let mut frames: Vec<Frame> = Vec::new();
122
123    for frame in &network.frames {
124        // New actors
125        for new_actor in &frame.new_actors {
126            let id = new_actor.actor_id.0;
127            let class_name = replay.objects
128                .get(new_actor.object_id.0 as usize)
129                .cloned()
130                .unwrap_or_default();
131
132            actor_class.insert(id, class_name.clone());
133
134            if class_name.contains("Car_Default") || class_name.contains("Car_Season") || class_name.contains("Car_PostGame") {
135                cars.insert(id, CarState {
136                    id,
137                    name: String::new(),
138                    team: 0,
139                    x: 0.0, y: 0.0, z: 0.0,
140                    yaw: 0.0,
141                    boost: 0,
142                });
143            }
144        }
145
146        // Deleted actors
147        for deleted in &frame.deleted_actors {
148            cars.remove(&deleted.0);
149        }
150
151        // Updated actors — each UpdatedActor has a single attribute
152        for replication in &frame.updated_actors {
153            let actor_id = replication.actor_id.0;
154            let class = actor_class.get(&actor_id).cloned().unwrap_or_default();
155
156            // Get attribute name from stream_id → objects table
157            let attr_name = replay.objects
158                .get(replication.stream_id.0 as usize)
159                .map(|s| s.as_str())
160                .unwrap_or("");
161
162            match &replication.attribute {
163                Attribute::RigidBody(rb) => {
164                    let pos = &rb.location;
165                    if class.contains("Ball") {
166                        ball_pos = Some(Vec3 { x: pos.x, y: pos.y, z: pos.z });
167                    } else if let Some(car) = cars.get_mut(&actor_id) {
168                        car.x = pos.x;
169                        car.y = pos.y;
170                        car.z = pos.z;
171                        let q = &rb.rotation;
172                        car.yaw = quat_to_yaw(q.x, q.y, q.z, q.w);
173                    }
174                }
175                Attribute::Byte(val) => {
176                    if attr_name.contains("ReplicatedBoost") {
177                        if let Some(car) = cars.get_mut(&actor_id) {
178                            car.boost = *val;
179                        }
180                    }
181                }
182                // PRI (Player Replication Info) links car → player
183                Attribute::Flagged { active, actor } => {
184                    if attr_name.contains("ReplicatedGameEvent") || attr_name.ends_with(":GameEvent") {
185                        // ignore game event links
186                    } else if attr_name.ends_with(":PRI") || attr_name.contains("PlayerReplicationInfo") {
187                        if *active {
188                            car_pri.insert(actor_id, actor.0);
189                        }
190                    }
191                }
192                // Player name on PRI actor
193                Attribute::String(s) => {
194                    if attr_name.ends_with(":PlayerName") {
195                        pri_name.insert(actor_id, s.clone());
196                    }
197                }
198                // Team on PRI actor (0 = blue, 1 = orange)
199                Attribute::Int(val) => {
200                    if attr_name.ends_with(":Team") {
201                        // team value is the team actor id; even = blue, odd = orange
202                        pri_team.insert(actor_id, (*val % 2).unsigned_abs() as u8);
203                    }
204                }
205                _ => {}
206            }
207        }
208
209        // Propagate name + team from PRI → car
210        for (car_id, pri_id) in &car_pri {
211            if let Some(car) = cars.get_mut(car_id) {
212                if let Some(name) = pri_name.get(pri_id) {
213                    car.name = name.clone();
214                }
215                if let Some(team) = pri_team.get(pri_id) {
216                    car.team = *team;
217                }
218            }
219        }
220
221        frames.push(Frame {
222            time: frame.time,
223            ball: ball_pos.clone(),
224            cars: cars.values().cloned().collect(),
225        });
226    }
227
228    let output = Output {
229        meta: ReplayMeta { team0_score, team1_score, duration, players },
230        frames,
231    };
232
233    println!("{}", serde_json::to_string(&output).unwrap());
234}
